  110.                  if(!empty($title) && $title!=$feed_row['title'])
    111. 

  111.                 {
    112. 

  112.                         $title = $db->EscapeString($title);
    113. 

  113.                         $url = $db->EscapeString($feed_row['url']);
    114. 

  114.                         $db->ExecSqlUpdate("UPDATE content_rss_aggregator_feeds SET title = '$title' WHERE url='$url'", false);
    115. 

  115.                         $this->refresh();
    116. 

  116.                 }
    117. 

  117.             }
    118. 

  118.         }
    119.

  176.          usually not homogenous.

  187.             $this->refresh();
