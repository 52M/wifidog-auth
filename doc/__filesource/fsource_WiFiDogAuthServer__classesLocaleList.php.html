  67.     private function getHumanLanguage($locale) {

  69.         $_retvalue = null;

  70.  
    71. 

  71.         $_humanLanguages = array('fr' => _("French"),
    72. 

  72.                                  'en' => _("English"),
    73. 

  73.                                  'de' => _("German"));
    74. 

  74.  
    75. 

  75.         if (array_key_exists($locale, $_humanLanguages)) {
    76. 

  76.             $_retvalue = $_humanLanguages[$locale];
    77. 

  77.         } else {
    78. 

  78.             $_retvalue = $locale;
    79. 

  79.         }

  85.      * Permet de générer un élément HTML select et de récupérer le résultat.

  93.     function GenererFormSelect($selectedClefPrimaire, $prefixeNomSelectUsager, $prefixeNomSelectObjet, $permetValeurNulle) {
    94. 

  94.         $retval = "";
    95. 

  95.         $resultats = "";
    96. 

  96.  
    97. 

  97.         $sql = "SELECT * FROM locales ORDER BY locales_id";
    98. 

  98.         $this->mBd->ExecSql($sql, $resultats, FALSE);
    99. 

  99.  
    100. 

  100.         $retval = "";
    101. 

  101.         $retval .= "<select name='$prefixeNomSelectUsager$prefixeNomSelectObjet'>\n";
    102. 

  102.  
    103. 

  103.         if ($permetValeurNulle == true) {
    104. 

  104.             $retval .= "<option value=''>---</option>\n";
    105. 

  105.         }
    106. 

  106.  
    107. 

  107.         while (list ($key, $value) = each($resultats)) {
    108. 

  108.             $retval .= "<option ";
    109. 

  109.  
    110. 

  110.             if ($value['locales_id'] == $selectedClefPrimaire || $selectedClefPrimaire == null && $selectedClefPrimaire == $this->GetDefault()) {
    111. 

  111.                 $retval .= 'selected="selected" ';
    112. 

  112.             }
    113. 

  113.  
    114. 

  114.             $retval .= "value='$value[locales_id]'>" . $this->getHumanLanguage($value["locales_id"]);
    115. 

  115.             $retval .= "</option>\n";
    116. 

  116.         }
    117. 

  117.  
    118. 

  118.         $retval .= "</select>\n";
    119. 

  119.  
    120. 

  120.         return $retval;
    121. 

  121.     }
