  72.     function getString() {
    73. 

  73.         // Init values
    74. 

  74.         $retval = null;
    75. 

  75.         $row = null;
    76. 

  76.         $_useCache = false;
    77. 

  77.         $_cachedData = null;
    78. 

  78.  
    79. 

  79.         // Create new cache objects
    80. 

  80.         $_cacheLanguage = new Cache('langstrings_' . $this->id . '_substring_' . substr(Locale :: getCurrentLocale()->getId(), 0, 2) . '_string', $this->id);
    81. 

  81.         $_cache = new Cache('langstrings_' . $this->id . '_substring__string', $this->id);
    82. 

  82.  
    83. 

  83.         // Check if caching has been enabled.
    84. 

  84.         if ($_cacheLanguage->isCachingEnabled) {
    85. 

  85.             if ($_cachedData = $_cacheLanguage->getCachedData()) {
    86. 

  86.                 // Return cached data.
    87. 

  87.                 $_useCache = true;
    88. 

  88.                 $retval = $_cachedData;
    89. 

  89.             } else {
    90. 

  90.                 // Language specific cached data has not been found.
    91. 

  91.                 // Try to get language independent cached data.
    92. 

  92.                 if ($_cachedData = $_cache->getCachedData()) {
    93. 

  93.                     // Return cached data.
    94. 

  94.                     $_useCache = true;
    95. 

  95.                     $retval = $_cachedData;
    96. 

  96.                 }
    97. 

  97.             }
    98. 

  98.         }
    99. 

  99.  
    100. 

  100.         if (!$_useCache) {
    101. 

  101.             //Get user's prefered language
    102. 

  102.             $sql = "SELECT value, locales_id, \n";
    103. 

  103.             $sql .= Locale :: getSqlCaseStringSelect(Locale :: getCurrentLocale()->getId());
    104. 

  104.             $sql .= " as score FROM langstring_entries WHERE langstring_entries.langstrings_id = '{$this->id}' AND value!='' ORDER BY score LIMIT 1";
    105. 

  105.             $this->mBd->ExecSqlUniqueRes($sql, $row, false);
    106. 

  106.  
    107. 

  107.             if ($row == null) {
    108. 

  108.                 $retval = "(Langstring vide)";
    109. 

  109.             } else {
    110. 

  110.                 $retval = $row['value'];
    111. 

  111.  
    112. 

  112.                 // Check if caching has been enabled.
    113. 

  113.                 if ($_cache->isCachingEnabled) {
    114. 

  114.                     // Save data into cache, because it wasn't saved into cache before.
    115. 

  115.                     $_cache->saveCachedData($retval);
    116. 

  116.                 }
    117. 

  117.             }
    118. 

  118.         }
    119. 

  119.  
    120. 

  120.         return $retval;
    121. 

  121.     }

  132.     function addString($string, $locale, $allow_empty_string = false) {
    133. 

  133.         $retval = false;
    134. 

  134.         $id = 'NULL';
    135. 

  135.         $idSQL = $id;
    136. 

  136.  
    137. 

  137.         if ($locale) {
    138. 

  138.             $language = new <a href="../WiFiDogAuthServer/Locale.html">Locale</a>($locale);
    139. 

  139.             $id = $language->GetId();
    140. 

  140.             $idSQL = "'".$id."'";
    141. 

  141.         }
    142. 

  142.  
    143. 

  143.         if ($allow_empty_string || ($string != null && $string != '')) {
    144. 

  144.             $string = $this->mBd->EscapeString($string);
    145. 

  145.             $this->mBd->ExecSqlUpdate("INSERT INTO langstring_entries (langstring_entries_id, langstrings_id, locales_id, value) VALUES ('".get_guid()."', '$this->id', $idSQL , '$string')", FALSE);
    146. 

  146.  
    147. 

  147.             // Create new cache object.
    148. 

  148.             $_cache = new Cache('langstrings_' . $this->id . '_substring_' .  $id . '_string', $this->id);
    149. 

  149.  
    150. 

  150.             // Check if caching has been enabled.
    151. 

  151.             if ($_cache->isCachingEnabled) {
    152. 

  152.                 // Remove old cached data.
    153. 

  153.                 $_cache->eraseCachedData();
    154. 

  154.  
    155. 

  155.                 // Save data into cache.
    156. 

  156.                 $_cache->saveCachedData($string);
    157. 

  157.             }
    158. 

  158.  
    159. 

  159.             $retval = true;
    160. 

  160.         }
    161. 

  161.  
    162. 

  162.         return $retval;
    163. 

  163.     }

  170.     function UpdateString($string, $locale) {
    171. 

  171.         // Init values
    172. 

  172.         $retval = false;
    173. 

  173.         $id = 'NULL';
    174. 

  174.         $row = null;
    175. 

  175.  
    176. 

  176.         if ($locale) {
    177. 

  177.             $language = new <a href="../WiFiDogAuthServer/Locale.html">Locale</a>($locale);
    178. 

  178.             $id = $language->GetId();
    179. 

  179.             $idSQL = "'" . $id . "'";
    180. 

  180.         }
    181. 

  181.  
    182. 

  182.         if ($string != null && $string != '') {
    183. 

  183.             $string = $this->mBd->EscapeString($string);
    184. 

  184.             // If the update returns 0 ( no update ), try inserting the record
    185. 

  185.             $this->mBd->ExecSqlUniqueRes("SELECT * FROM langstring_entries WHERE locales_id = $idSQL AND langstrings_id = '$this->id'", $row, false);
    186. 

  186.  
    187. 

  187.             if ($row != null) {
    188. 

  188.                 $this->mBd->ExecSqlUpdate("UPDATE langstring_entries SET value = '$string' WHERE langstrings_id = '$this->id' AND locales_id = $idSQL", false);
    189. 

  189.  
    190. 

  190.                 // Create new cache object.
    191. 

  191.                 $_cache = new Cache('langstrings_' . $this->id . '_substring_' .  $id . '_string', $this->id);
    192. 

  192.  
    193. 

  193.                 // Check if caching has been enabled.
    194. 

  194.                 if ($_cache->isCachingEnabled) {
    195. 

  195.                     // Remove old cached data.
    196. 

  196.                     $_cache->eraseCachedData();
    197. 

  197.  
    198. 

  198.                     // Save data into cache.
    199. 

  199.                     $_cache->saveCachedData($string);
    200. 

  200.                 }
    201. 

  201.             } else {
    202. 

  202.                 $this->addString($string, $locale);
    203. 

  203.             }
    204. 

  204.  
    205. 

  205.             $retval = true;
    206. 

  206.         }
    207. 

  207.  
    208. 

  208.         return $retval;
    209. 

  209.     }
