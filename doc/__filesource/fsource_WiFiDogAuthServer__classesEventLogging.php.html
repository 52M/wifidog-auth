  35. /**
    36. 

  36.  * EventLogging: a message logging class.
    37. 

  37.  *
    38. 

  38.  * Logging messages of various levels, from debug to critical, from php or
    39. 

  39.  * generated by the program, can be formatted and directed to a file, be
    40. 

  40.  * displayed on the html page, as comments or as a table, or be held and
    41. 

  41.  * collected for display later in a special <div> block.
    42. 

  42.  *
    43. 

  43.  * Logging messages once accepted at intake will be processed by all
    44. 

  44.  * active output channels.  ie, the message will be written by all
    45. 

  45.  * the output channels, not just one.
    46. 

  46.  *
    47. 

  47.  * Output channels currently supported:
    48. 

  48.  *
    49. 

  49.  * FileChannel - appends messages to a file
    50. 

  50.  * ArrayChannel - accumulates messages in an array
    51. 

  51.  * NullChannel - discards all messages
    52. 

  52.  * PrintChannel - writes messages via "echo" or "print"
    53. 

  53.  *
    54. 

  54.  * Output formats currently supported:
    55. 

  55.  * SyslogFormatter - like syslog, with a datestamp prefixed
    56. 

  56.  * HTMLFormatter - as table row elements
    57. 

  57.  * HTMLCommentsFormatter - as html comments
    58. 

  58.  *
    59. 

  59.  * Collection of logging messages can be turned off temporarily,
    60. 

  60.  * either by using the standard php '@' prefix, or by using the
    61. 

  61.  * off() and on() methods, and their static equivalents,
    62. 

  62.  * EventLogging::stOn() and EventLogging::stOff().
    63. 

  63.  *
    64. 

  64.  * Debugging of specific methods, functions, or files is supported
    65. 

  65.  * by allowing messages from such contexts to be redirected to a
    66. 

  66.  * unique channel, or the message can be catapulted to a different
    67. 

  67.  * importance level, like critical, to make sure it gets written by
    68. 

  68.  * one of the output channels.
    69. 

  69.  *
    70. 

  70.  * By using the addContextFilter method you can specify special
    71. 

  71.  * treatment for messages based on the method, function or file the
    72. 

  72.  * message came from.  debug_backtrace() is used to see if the context
    73. 

  73.  * is on the stack.  If it is, then the special treatment is applied.
    74. 

  74.  *
    75. 

  75.  * @todo Extensible to a more general transactional processing system,
    76. 

  76.  *  useful in a context where the work to be done should have BEGIN
    77. 

  77.  *  WORK and COMMIT/ROLLBACK logic, but either the database in use
    78. 

  78.  *  does not support it, or the work being done is multiply facited,
    79. 

  79.  *  involving non-database elements that should be rolled back too.
    80. 

  80.  *
    81. 

  81.  *
    82. 

  82.  *  usage: at the start in mainline, to setup the logging environment
    83. 

  83.  *
    84. 

  84.  *  $log = EventLogging::stGlobal($_ENV['DEBUG']);
    85. 

  85.  *
    86. 

  86.  *  Examples of strings to pass stGlobal:
    87. 

  87.  *    "Class::Method(critical)" - matches messages from Class::Method, makes them critical
    88. 

  88.  *    "Class::Method(off)" - matches messages from Class::Method, dumps them
    89. 

  89.  *    "strict~/var:\sDeprecated/(off)" - matches message text with "strict" level, dumps them
    90. 

  90.  *    "Function(critical) /this/path/file.php(debug)" - matches messages from Function, makes them critical,
    91. 

  91.  *       and matches any messages by any code in /this/path/file.php, makes them debug level
    92. 

  92.  *  Separate multiple entries with spaces, not commas.
    93. 

  93.  *
    94. 

  94.  *  Add output channels, globally.  All messages will pass to these channels, unless
    95. 

  95.  *  a function/method/file override comes into play.
    96. 

  96.  *
    97. 

  97.  *  $log->addChannel(new FileChannel("logg.txt", new SyslogFormatter(), 'error,notice'));
    98. 

  98.  *  $log->addChannel(new PrintChannel(new HTMLCommentsFormatter(), '=debug'));
    99. 

  99.  *  $log->addChannel(new ArrayChannel(new HTMLFormatter(), 'warning,notice'));
    100. 

  100.  *
    101. 

  101.  *  Or, start up logging all at once:
    102. 

  102.  *  EventLogging::SetupErrorHandling('notice~/Constant.+already\sdefined/(off)',
    103. 

  103.  *      new PrintChannel(new HTMLCommentsFormatter(), '=debug'),
    104. 

  104.  *      new ArrayChannel(new HTMLFormatter(), 'all,!debug'));
    105. 

  105.  *
    106. 

  106.  *
    107. 

  107.  *  automatic intake/capture of system messages and exceptions
    108. 

  108.  *  elLogMessage("message", level)
    109. 

  109.  *  elLogError("message")
    110. 

  110.  *  elLogWarning("message")
    111. 

  111.  *  elLogNotice("message")
    112. 

  112.  *  elLogInfo("message")
    113. 

  113.  *  elLogDebug("message")
    114. 

  114.  *
    115. 

  115.  *  or $log->message("message", EventObject::EVO_USER_WARNING)
    116. 

  116.  *  or EventLogging::stMessage("message", EventObject::EVO_USER_WARNING)
    117. 

  117.  *
    118. 

  118.  *  or $log->error("message")
    119. 

  119.  *  or $log->warning("message")
    120. 

  120.  *  or $log->notice("message")
    121. 

  121.  *  or $log->info("message")
    122. 

  122.  *  or $log->debug("message")
    123. 

  123.  *
    124. 

  124.  *  broadcastEvent(new Event( ... ))
    125. 

  125.  *  elLogEvent()
    126. 

  126.  *
    127. 

  127.  *  access to the logging stuff is easy, global
    128. 

  128.  *  $log = EventLogging::getGlobal();
    129. 

  129.  *
    130. 

  130.  *  Howto silence logging:
    131. 

  131.  *    $log->off();
    132. 

  132.  *    or EventLogging::stOff();
    133. 

  133.  *    or put "@" in front of the expression (standard php trick)
    134. 

  134.  *
    135. 

  135.  *    then "blah blah blah stuff that emits annoying errors and won't shut up"
    136. 

  136.  *
    137. 

  137.  *    $log->on();
    138. 

  138.  *    or EventLogging::stOn();
    139. 

  139.  *
    140. 

  140.  *  Howto zero in on buggy routines, or in development
    141. 

  141.  *
    142. 

  142.  *  Context filters are supposed to be applied at the global level,
    143. 

  143.  *  during the initialization phase.  They allow file, class or
    144. 

  144.  *  function overrides on logging behaviour.  Using context filters
    145. 

  145.  *  you can put all your special logging stuff in one place, rather
    146. 

  146.  *  than tracking down the subroutine you want to tweak and editing
    147. 

  147.  *  it.
    148. 

  148.  *
    149. 

  149.  *  Note that the effectiveLevel controls how the log message passes
    150. 

  150.  *  through the channels, in that the channels typically screen out
    151. 

  151.  *  messages based on their level.
    152. 

  152.  *
    153. 

  153.  *  The formatting of the message for display is taken to be a
    154. 

  154.  *  separate thing, so that you can override the effectiveLevel to
    155. 

  155.  *  critical so that the message will be printed by most channels,
    156. 

  156.  *  while not changing the printed message, so that it still says
    157. 

  157.  *  "debug" or whatever.
    158. 

  158.  *
    159. 

  159.  *  EventLogging::stAddContextFilter(
    160. 

  160.  *      array('class' => 'ClassName',
    161. 

  161.  *            'function' => 'MethodName',
    162. 

  162.  *            'effectiveLevel'=>'critical'));
    163. 

  163.  *
    164. 

  164.  *  EventLogging::stAddContextFilter(
    165. 

  165.  *      array('function' => 'FunctionName',
    166. 

  166.  *            'redirect'=>new PrintChannel(new HTMLCommentsFormatter(), 'debug')));
    167. 

  167.  *
    168. 

  168.  *  EventLogging::stAddContextFilter(
    169. 

  169.  *      array('file' => 'FileName',
    170. 

  170.  *            'redirect'=>new NullChannel('debug', 'blockPercolation')));
    171. 

  171.  *
    172. 

  172.  *  This will do the same sort of thing as a context filter, but you
    173. 

  173.  *  have to put it in the method or function:
    174. 

  174.  *
    175. 

  175.  *  $log = new EventLogging(); // knows it's a context change, context removed when $log __destruct is called
    176. 

  176.  *  $log->setEffectiveLevel('critical');
    177. 

  177.  *
    178. 

  178.  * @package    WiFiDogAuthServer

    183.

  284.   public function broadcastEvent($pEvent, $pOn=null, $pDebug=null, $pEffectiveImportance=null) {
    285. 

  285.     if (isset($this->on)) $pOn = $this->on;
    286. 

  286.     if (isset($this->debug)) $pDebug = $this->debug;
    287. 

  287.     if (isset($this->effectiveLevel)) $pEffectiveImportance = $this->effectiveLevel;
    288. 

  288.  
    289. 

  289.     $myBlockPercolation = false;
    290. 

  290.  
    291. 

  291.     $myCountSinks = 0;
    292. 

  292.  
    293. 

  293.     $myWatchpoint = null;
    294. 

  294.     $myForceOnEvent = null;
    295. 

  295.     if (!empty($this->contextFilters) && count($this->contextFilters)>0) {
    296. 

  296.       // logging is supposed to be quick.  this is slow, but is most useful for debugging.
    297. 

  297.       $myCountSinks++;
    298. 

  298.  
    299. 

  299.       $myWatchpoint = $this->filterContext($pEvent); // returns true or false or a channel object
    300. 

  300.  
    301. 

  301.       $myForceOnEvent = $pEvent->getOnFlag();
    302. 

  302.       if (isset($myForceOnEvent)) $pOn = $myForceOnEvent;
    303. 

  303.  
    304. 

  304.       if (is_object($myWatchpoint))
    305. 

  305.     if ($myWatchpoint->receiveEvent($pEvent)) {
    306. 

  306.       // since this is a specific match at this level
    307. 

  307.       // allow this section here the ability to override the more general stuff
    308. 

  308.       // ie, if this specific match results in a "don't let any other channels run" flag, do it immediately
    309. 

  309.       $myBlockPercolation = true;
    310. 

  310.       return $myBlockPercolation;
    311. 

  311.     }
    312. 

  312.     }
    313. 

  313.  
    314. 

  314.     if (!empty($this->stackPtrMoreSpecific)) {
    315. 

  315.       $myCountSinks++;
    316. 

  316.       $myBlockPercolation = $this->stackPtrMoreSpecific->broadcastEvent($pEvent, $pOn, $pDebug, $pEffectiveImportance);
    317. 

  317.     }
    318. 

  318.  
    319. 

  319.     if ((error_reporting()==0 && empty($pDebug)) || $myBlockPercolation)
    320. 

  320.       return $myBlockPercolation;
    321. 

  321.  
    322. 

  322.     if (!isset($pOn) || $pOn) {
    323. 

  323.       $myWatchers =& $this->observers;
    324. 

  324.  
    325. 

  325.       if (count($myWatchers)==0 && $this===self::$globalTracker && $myCountSinks==0) {
    326. 

  326.     // the out-of-the box behaviour of EventLogging should be the same as PHP
    327. 

  327.     if (empty($this->defaultObserver))
    328. 

  328.       $this->defaultObserver = new PrintChannel(new HTMLFormatter(), 'error,notice');
    329. 

  329.  
    330. 

  330.     if (!empty($this->defaultObserver))
    331. 

  331.       $myWatchers = array($this->defaultObserver);
    332. 

  332.       }
    333. 

  333.  
    334. 

  334.       foreach ($myWatchers as $watcher) {
    335. 

  335.     $myCountSinks++;
    336. 

  336.     if ($watcher->receiveEvent($pEvent, $pEffectiveImportance))
    337. 

  337.       $myBlockPercolation = true;;
    338. 

  338.       }
    339. 

  339.     }
    340. 

  340.  
    341. 

  341.     return $myBlockPercolation;
    342. 

  342.   }

  410.   public function isOn() {
    411. 

  411.     if (error_reporting()==0 && empty($this->debug)) return false;
    412.

  684.     public function quickContextFilters($pString) {
    685. 

  685.     if (empty($pString)) return 0;
    686. 

  686.  
    687. 

  687.     $myCounter = 0;
    688. 

  688.     $myList = preg_split('/\s+/', $pString, -1, PREG_SPLIT_NO_EMPTY);
    689. 

  689.     foreach ($myList as $myItem) {
    690. 

  690.       if (preg_match('/^([^()]+)\(([^()]+)\)$/', $myItem, $myMatches)) {
    691. 

  691.     $myContext = $myMatches[1];
    692. 

  692.     $myOverride = $myMatches[2];
    693. 

  693.     $filter = array();
    694. 

  694.     if (strpos($myContext, '::') !== false) {
    695. 

  695.       // class
    696. 

  696.       $tmp = explode('::', $myContext, 2);
    697. 

  697.       $filter['class'] = trim($tmp[0]);
    698. 

  698.       if (preg_match('/\S/', $tmp[1])) $filter['function'] = trim($tmp[1]);
    699. 

  699.       $filter['effectiveLevel'] = trim($myOverride);
    700. 

  700.     }
    701. 

  701.     elseif (strpos($myContext, '~') !== false) {
    702. 

  702.       // message match
    703. 

  703.       $tmp = explode('~', $myContext, 2);
    704. 

  704.       $filter['matchlevel'] = trim($tmp[0]);
    705. 

  705.       $filter['matchpattern'] = trim($tmp[1]);
    706. 

  706.       $filter['effectiveLevel'] = trim($myOverride);
    707. 

  707.     }
    708. 

  708.     elseif (strpos($myContext, '/') !== false) {
    709. 

  709.       // file
    710. 

  710.       $filter['file'] = trim($myContext);
    711. 

  711.       $filter['effectiveLevel'] = trim($myOverride);
    712. 

  712.     }
    713. 

  713.     else {
    714. 

  714.       // function
    715. 

  715.       $filter['function'] = trim($myContext);
    716. 

  716.       $filter['effectiveLevel'] = trim($myOverride);
    717. 

  717.     }
    718. 

  718.  
    719. 

  719.     if (EventObject::stClassifyErrorType($myOverride)) {
    720. 

  720.       EventLogging::stAddContextFilter($filter);
    721. 

  721.       $myCounter++;
    722. 

  722.     }
    723. 

  723.       }
    724. 

  724.     }
    725. 

  725.  
    726. 

  726.     return $myCounter;
