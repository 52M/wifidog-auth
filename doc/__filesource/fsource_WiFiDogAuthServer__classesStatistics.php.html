  280.     /** Get the actual SQL fragment to get the candidates rows from the connections table,
    281. 

  281.      * once obeying all the report configuration constraints.  Only connections
    282. 

  282.      * with actuall data transferred is considered.  Connections is always
    283. 

  283.      * joined to the nodes table, but not to network or users.
    284. 

  284.      * @param $select_columns The selected columns, will be inserted between
    285. 

  285.      * between SELECT and FROM
    286. 

  286.      * @param $join_users true or false, Should we join with the users table?
    287. 

  287.      * @return SQL select statemnt.  You can append additional AND and GROUP BY
    288. 

  288.      * clauses */
    289. 

  289.     public function getSqlCandidateConnectionsQuery($select_columns = '*', $join_users = false)
    290. 

  290.     {
    291. 

  291.         $sql = '';
    292. 

  292.         $date_constraint = $this->getSqlDateConstraint('timestamp_in');
    293. 

  293.         $node_constraint = $this->getSqlNodeConstraint('connections.node_id');
    294. 

  294.         $network_constraint = $this->getSqlNetworkConstraint('nodes.network_id');
    295. 

  295.         $user_constraint = $this->getSqlUserConstraint();
    296. 

  296.         $join_users_sql = '';
    297. 

  297.         if ($join_users)
    298. 

  298.         {
    299. 

  299.             $join_users_sql = "JOIN users ON (connections.user_id = users.user_id)";
    300. 

  300.         }
    301. 

  301.         $sql .= "SELECT $select_columns \n";
    302. 

  302.  
    303. 

  303.         $sql .= "FROM connections  \n";
    304. 

  304.         $sql .= "JOIN nodes ON (connections.node_id = nodes.node_id) \n";
    305. 

  305.         $sql .= "$join_users_sql \n";
    306. 

  306.         $sql .= "WHERE (incoming!=0 OR outgoing!=0) \n";
    307. 

  307.         $sql .= " {$date_constraint} {$node_constraint} {$network_constraint} {$user_constraint}";
    308. 

  308.         return $sql;
    309. 

  309.     }
