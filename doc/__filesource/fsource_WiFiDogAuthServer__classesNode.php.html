  104.     public static function getCurrentRealNode()
    105. 

  105.     {
    106. 

  106.         global $db;
    107. 

  107.         $retval = null;
    108. 

  108.         $sql = "SELECT node_id, last_heartbeat_ip from nodes WHERE last_heartbeat_ip='$_SERVER[REMOTE_ADDR]' ORDER BY last_heartbeat_timestamp DESC";
    109. 

  109.         $node_rows = null;
    110. 

  110.         $db->ExecSql($sql, $node_rows, false);
    111. 

  111.         $num_match = count($node_rows);
    112. 

  112.         if ($num_match == 0)
    113. 

  113.         {
    114. 

  114.  
    115. 

  115.             // User is not physically connected to a node
    116. 

  116.             $retval = null;
    117. 

  117.         }
    118. 

  118.         else
    119. 

  119.             if ($num_match = 1)
    120. 

  120.             {
    121. 

  121.                 // Only a single node matches, the user is presumed to be there
    122. 

  122.                 $retval = new self($node_rows[0]['node_id']);
    123. 

  123.             }
    124. 

  124.             else
    125. 

  125.             {
    126. 

  126.                 /* We have more than one node matching the IP (the nodes are behind the same NAT).
    127. 

  127.                  * We will try to discriminate by finding which node the user last authenticated against.
    128. 

  128.                  * If the IP matches, we can be pretty certain the user is there.
    129. 

  129.                  */
    130. 

  130.                 $retval = null;
    131. 

  131.                 $current_user = User :: getCurrentUser();
    132. 

  132.                 if ($current_user != null)
    133. 

  133.                 {
    134. 

  134.                     $current_user_id = $current_user->getId();
    135. 

  135.                     $_SERVER['REMOTE_ADDR'];
    136. 

  136.                     $sql = "SELECT node_id, last_heartbeat_ip from connections NATURAL JOIN nodes WHERE user_id='$current_user_id' ORDER BY last_updated DESC ";
    137. 

  137.                     $db->ExecSql($sql, $node_rows, false);
    138. 

  138.                     $node_row = $node_rows[0];
    139. 

  139.                     if ($node_row != null && $node_row['last_heartbeat_ip'] == $_SERVER['REMOTE_ADDR'])
    140. 

  140.                     {
    141. 

  141.                         $retval = new self($node_row['node_id']);
    142. 

  142.                     }
    143. 

  143.                 }
    144. 

  144.             }
    145. 

  145.         return $retval;
    146. 

  146.     }
