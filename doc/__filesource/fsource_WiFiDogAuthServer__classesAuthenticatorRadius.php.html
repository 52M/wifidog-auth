  95.     /** Attempts to login a user against the authentication source.  If successfull, returns a User object
    96. 

  96.      * @param username:  A valid identifying token for the source.  Not necessarily unique.  For local user, bots username and email are valid.
    97. 

  97.      * @param password:  Clear text password.
    98. 

  98.      * @retval The actual User object if login was successfull, false otherwise.
    99. 

  99.      */
    100. 

  100.     function login($username, $password, & $errmsg = null)
    101. 

  101.     {
    102. 

  102.         global $db;
    103. 

  103.         $security = new Security();
    104. 

  104.         $retval = false;
    105. 

  105.         $username = $db->EscapeString($username);
    106. 

  106.         $password = $db->EscapeString($password);
    107. 

  107.  
    108. 

  108.         /*
    109. 

  109.          * Supported encryption methods are :
    110. 

  110.          *
    111. 

  111.          * CHAP_MD5 :Challenge-Handshake Authentication Protocol with MD5
    112. 

  112.          * MSCHAPv1 and MSCHAPv2 : Microsoft's CHAP implementation
    113. 

  113.          */
    114. 

  114.         switch ($this->mRadius_encryption_method)
    115. 

  115.         {
    116. 

  116.             case "PAP" :
    117. 

  117.             case "CHAP_MD5" :
    118. 

  118.             case "MSCHAPv1" :
    119. 

  119.             case "MSCHAPv2" :
    120. 

  120.                 // Instanciate PEAR class
    121. 

  121.                 $classname = 'Auth_RADIUS_'.$this->mRadius_encryption_method;
    122. 

  122.                 $radius_server = new $classname ($username, $password);
    123. 

  123.                 $radius_server->addServer($this->mRadius_hostname, $this->mRadius_auth_port, $this->mRadius_secret_key);
    124. 

  124.                 break;
    125. 

  125.             default :
    126. 

  126.                 // Invalid encryption method
    127. 

  127.                 $errmsg = _("Invalid RADIUS encryption method.");
    128. 

  128.                 return false;
    129. 

  129.         }
    130. 

  130.  
    131. 

  131.         // Instructing PEAR RADIUS class auth parameters
    132. 

  132.         $radius_server->username = $username;
    133. 

  133.         // Depending on the auth method, generate challenge response
    134. 

  134.         switch ($this->mRadius_encryption_method)
    135. 

  135.         {
    136. 

  136.             case 'CHAP_MD5' :
    137. 

  137.             case 'MSCHAPv1' :
    138. 

  138.                 $classname = $this->mRadius_encryption_method == 'MSCHAPv1' ? 'Crypt_CHAP_MSv1' : 'Crypt_CHAP_MD5';
    139. 

  139.                 $crypt_class = new $classname;
    140. 

  140.                 $crypt_class->password = $password;
    141. 

  141.                 $radius_server->challenge = $crypt_class->challenge;
    142. 

  142.                 $radius_server->chapid = $crypt_class->chapid;
    143. 

  143.                 $radius_server->response = $crypt_class->challengeResponse();
    144. 

  144.                 $radius_server->flags = 1;
    145. 

  145.                 break;
    146. 

  146.             case 'MSCHAPv2' :
    147. 

  147.                 $crypt_class = new Crypt_CHAP_MSv2;
    148. 

  148.                 $crypt_class->username = $username;
    149. 

  149.                 $crypt_class->password = $password;
    150. 

  150.                 $radius_server->challenge = $crypt_class->authChallenge;
    151. 

  151.                 $radius_server->peerChallenge = $crypt_class->peerChallenge;
    152. 

  152.                 $radius_server->chapid = $crypt_class->chapid;
    153. 

  153.                 $radius_server->response = $crypt_class->challengeResponse();
    154. 

  154.                 break;
    155. 

  155.             default :
    156. 

  156.                 $radius_server->password = $password;
    157. 

  157.                 break;
    158. 

  158.         }
    159. 

  159.  
    160. 

  160.         if (!$radius_server->start())
    161. 

  161.         {
    162. 

  162.             $errmsg = _("Could not initiate PEAR RADIUS Auth class : ".$radius_server->getError());
    163. 

  163.             return false;
    164. 

  164.         }
    165. 

  165.  
    166. 

  166.         // Send the authentication request to the RADIUS server
    167. 

  167.         $result = $radius_server->send();
    168. 

  168.  
    169. 

  169.         if (PEAR :: isError($result))
    170. 

  170.         {
    171. 

  171.             $errmsg = _("Failed to send authentication request to the RADIUS server. : ".$result->getMessage());
    172. 

  172.             return false;
    173. 

  173.         }
    174. 

  174.         else
    175. 

  175.             if ($result === true)
    176. 

  176.             {
    177. 

  177.                 // RADIUS authentication succeeded !
    178. 

  178.                 // Now checking for local copy of this user
    179. 

  179.                 $user_info = null;
    180. 

  180.                 $sql = "SELECT user_id, pass FROM users WHERE (username='$username') AND account_origin='".$this->getNetwork()->getId()."'";
    181. 

  181.                 $db->ExecSqlUniqueRes($sql, $user_info, false);
    182. 

  182.  
    183. 

  183.                 if ($user_info != null)
    184. 

  184.                 {
    185. 

  185.                     $user = new User($user_info['user_id']);
    186. 

  186.                     if ($user->isUserValid($errmsg))
    187. 

  187.                     {
    188. 

  188.                         $retval = & $user;
    189. 

  189.                         User :: setCurrentUser($user);
    190. 

  190.                         $errmsg = _("Login successfull");
    191. 

  191.                     }
    192. 

  192.                     else
    193. 

  193.                     {
    194. 

  194.                         $retval = false;
    195. 

  195.                         //Reason for refusal is already in $errmsg
    196. 

  196.                     }
    197. 

  197.                 }
    198. 

  198.                 else
    199. 

  199.                 {
    200. 

  200.                     // This user has been succcessfully authenticated through remote RADIUS, but it's not yet in our local database
    201. 

  201.                     // Creating the user with a Global Unique ID, empty email and password
    202. 

  202.                     // Local database password hashing is based on an empty string ( we do not store remote passwords )
    203. 

  203.                     $user = User :: createUser(get_guid(), $username, $this->getNetwork(), "", "");
    204. 

  204.                     $retval = & $user;
    205. 

  205.                     // Validate the user right away !
    206. 

  206.                     $user->setAccountStatus(ACCOUNT_STATUS_ALLOWED);
    207. 

  207.                     User :: setCurrentUser($user);
    208. 

  208.                     $errmsg = _("Login successfull");
    209. 

  209.                 }
    210. 

  210.                 return $retval;
    211. 

  211.             }
    212. 

  212.             else
    213. 

  213.             {
    214. 

  214.                 $errmsg = _("The RADIUS server rejected this username/password combination.");
    215. 

  215.                 return false;
    216. 

  216.             }
    217. 

  217.  
    218. 

  218.         $radius_server->close();
    219. 

  219.     }

  221.     /** Start accounting traffic for the user
    222. 

  222.      * $conn_id:  The connection id for the connection to work on */
    223. 

  223.  
    224. 

  224.     function acctStart($conn_id, & $errmsg = null)
    225. 

  225.     {
    226. 

  226.         global $db;
    227. 

  227.         $info = null;
    228. 

  228.         $conn_id = $db->EscapeString($conn_id);
    229. 

  229.         $db->ExecSqlUniqueRes("SELECT NOW(), *, CASE WHEN ((NOW() - reg_date) > networks.validation_grace_time) THEN true ELSE false END AS validation_grace_time_expired FROM connections JOIN users ON (users.user_id=connections.user_id) JOIN networks ON (users.account_origin = networks.network_id) WHERE connections.conn_id=$conn_id", $info, false);
    230. 

  230.  
    231. 

  231.         // RADIUS accounting start
    232. 

  232.         $radius_acct = new Auth_RADIUS_Acct_Start;
    233. 

  233.         $radius_acct->addServer($this->mRadius_hostname, $this->mRadius_acct_port, $this->mRadius_secret_key);
    234. 

  234.         // Specify the user for which accounting will be done
    235. 

  235.         $radius_acct->username = $info['username'];
    236. 

  236.         // Specify the way the user has been authenticated ( via RADIUS, the class did it )
    237. 

  237.         $radius_acct->authentic = RADIUS_AUTH_RADIUS;
    238. 

  238.         // Set the session ID to the generated token
    239. 

  239.         $radius_acct->session_id = $info['token'];
    240. 

  240.  
    241. 

  241.         $status = $radius_acct->start();
    242. 

  242.         if (PEAR :: isError($status))
    243. 

  243.             return false;
    244. 

  244.  
    245. 

  245.         $result = $radius_acct->send();
    246. 

  246.         if (PEAR :: isError($result))
    247. 

  247.         {
    248. 

  248.             $errmsg = "Could not send accounting request to RADIUS server.";
    249. 

  249.             return false;
    250. 

  250.         }
    251. 

  251.         else
    252. 

  252.             if ($result !== true)
    253. 

  253.             {
    254. 

  254.                 $radius_acct->close();
    255. 

  255.                 $errmsg = "Accounting request rejected by RADIUS server.";
    256. 

  256.                 return false;
    257. 

  257.             }
    258. 

  258.  
    259. 

  259.         $radius_acct->close();
    260. 

  260.  
    261. 

  261.         // Run generic accounting ( local traffic counters ) only if RADIUS went OK
    262. 

  262.         parent :: acctStart($conn_id);
    263. 

  263.         return true;
    264. 

  264.     }

  266.     /** Update traffic counters
    267. 

  267.      * $conn_id: The connection id for the connection to work on */

  268.  
    269. 

  269.     function acctUpdate($conn_id, $incoming, $outgoing, & $errmsg = null)
    270. 

  270.     {
    271. 

  271.         // Call generic traffic updater ( local database )
    272. 

  272.         parent :: acctUpdate($conn_id, $incoming, $outgoing);
    273. 

  273.         global $db;
    274. 

  274.         $info = null;
    275. 

  275.         $conn_id = $db->EscapeString($conn_id);
    276. 

  276.         $db->ExecSqlUniqueRes("SELECT NOW(), *, CASE WHEN ((NOW() - reg_date) > networks.validation_grace_time) THEN true ELSE false END AS validation_grace_time_expired FROM connections JOIN users ON (users.user_id=connections.user_id) JOIN networks ON (users.account_origin = networks.network_id) WHERE connections.conn_id=$conn_id", $info, false);
    277. 

  277.  
    278. 

  278.         // RADIUS accounting ping
    279. 

  279.         // Session is completely based on Database time
    280. 

  280.         $session_time = strtotime($info['now']) - strtotime($info['timestamp_in']);
    281. 

  281.  
    282. 

  282.         $radius_acct = new Auth_RADIUS_Acct_Update;
    283. 

  283.         $radius_acct->addServer($this->mRadius_hostname, $this->mRadius_acct_port, $this->mRadius_secret_key);
    284. 

  284.         // Specify the user for which accounting will be done
    285. 

  285.         $radius_acct->username = $info['username'];
    286. 

  286.         $racct->session_time = $session_time;
    287. 

  287.         // Set the session ID to the generated token
    288. 

  288.         $radius_acct->session_id = $info['token'];
    289. 

  289.  
    290. 

  290.         $status = $radius_acct->start();
    291. 

  291.         if (PEAR :: isError($status))
    292. 

  292.             return false;
    293. 

  293.  
    294. 

  294.         // Send traffic data along with the request
    295. 

  295.         $radius_acct->putAttribute(RADIUS_ACCT_INPUT_PACKETS, $incoming);
    296. 

  296.         $radius_acct->putAttribute(RADIUS_ACCT_OUTPUT_PACKETS, $outgoing);
    297. 

  297.  
    298. 

  298.         $result = $radius_acct->send();
    299. 

  299.         if (PEAR :: isError($result))
    300. 

  300.         {
    301. 

  301.             $errmsg = "Could not send accounting request to RADIUS server.";
    302. 

  302.             return false;
    303. 

  303.         }
    304. 

  304.         else
    305. 

  305.             if ($result !== true)
    306. 

  306.             {
    307. 

  307.                 $radius_acct->close();
    308. 

  308.                 $errmsg = "Accounting request rejected by RADIUS server.";
    309. 

  309.                 return false;
    310. 

  310.             }
    311. 

  311.  
    312. 

  312.         $radius_acct->close();
    313. 

  313.         return true;
    314. 

  314.     }
