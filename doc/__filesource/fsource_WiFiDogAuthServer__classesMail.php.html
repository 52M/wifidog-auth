  61.     private function encodeMimeHeader($header)
    62. 

  62.     {
    63. 

  63.         // BASE 64 according to the RFC
    64. 

  64.         // Taken from : www.php.net mb_send_mail comments
    65. 

  65.         $header = preg_replace('/([^a-z ])/ie', 'sprintf("=%02x",ord(StripSlashes("\\1")))', $header);
    66. 

  66.         $header = str_replace(' ', '_', $header);
    67. 

  67.         return "=?utf-8?Q?$header?=";
