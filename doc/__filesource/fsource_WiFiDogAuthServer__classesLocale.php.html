  81.     function __construct($p_locale) {
    82. 

  82.         $matches = self :: decomposeLocaleId($p_locale);
    83. 

  83.         $locale = $matches[1];
    84. 

  84.         $this->mLang = $matches[1];
    85. 

  85.  
    86. 

  86.         if ($this->mLang == null) {
    87. 

  87.             throw new Exception(_("Locale(): Impossible de trouver une langue correspondant à $locale"), EXCEPTION_CREATE_OBJECT_FAILED);
    88. 

  88.         }
    89. 

  89.  
    90. 

  90.         if (empty ($matches[2])) {
    91. 

  91.             $this->mPays = null;
    92. 

  92.         } else {
    93. 

  93.             $locale .= '_'.$matches[2];
    94. 

  94.             $this->mPays = $matches[2];
    95. 

  95.         }
    96. 

  96.  
    97. 

  97.         if (empty ($matches[3])) {
    98. 

  98.             // TODO: Support subcode.
    99. 

  99.         } else {
    100. 

  100.             $locale .= '_'.$matches[3];
    101. 

  101.         }
    102. 

  102.  
    103. 

  103.         $this->mId = $locale;
    104. 

  104.     }

  115.     public static function getCurrentLocale() {
    116. 

  116.         global $session;
    117. 

  117.         $object = null;
    118. 

  118.         $locale_id = $session->get(SESS_LANGUAGE_VAR);
    119. 

  119.  
    120. 

  120.         /* Try to guess the lang */
    121. 

  121.         if (empty($locale_id)) {
    122. 

  122.             $locale_id = self :: getBestLanguage();
    123. 

  123.         }
    124. 

  124.  
    125. 

  125.         /* If we still don't have it, fill in default */
    126. 

  126.         if (empty ($locale_id)) {
    127. 

  127.             $object = self :: getObject(DEFAULT_LANG);
    128. 

  128.             self :: setCurrentLocale($object);
    129. 

  129.         } else {
    130. 

  130.             $object = self :: getObject($locale_id);
    131. 

  131.             self :: setCurrentLocale($object);
    132. 

  132.         }
    133. 

  133.  
    134. 

  134.         return $object;
    135. 

  135.     }

  138.       * Try to find best language according to HTTP_ACCEPT_LANGUAGE passed

  143.     public static function getBestLanguage() {
    144. 

  144.         global $AVAIL_LOCALE_ARRAY;
    145. 

  145.  
    146. 

  146.         if (defined($_SERVER["HTTP_ACCEPT_LANGUAGE"])) {
    147. 

  147.             foreach(split(';', $_SERVER["HTTP_ACCEPT_LANGUAGE"]) as $lang) {
    148. 

  148.                 foreach($AVAIL_LOCALE_ARRAY as $avail_lang => $lang_description) {
    149. 

  149.                     $lang = ereg_replace("[_-].*$", "", $lang);
    150. 

  150.                     $avail_lang_trimmed = ereg_replace("[_-].*$", "", $avail_lang);
    151. 

  151.  
    152. 

  152.                     if ($lang == $avail_lang_trimmed) {
    153. 

  153.                         return $avail_lang;
    154. 

  154.                     }
    155. 

  155.                 }
    156. 

  156.             }
    157. 

  157.         }

  165.     public static function setCurrentLocale($locale) {
    166. 

  166.         global $session;
    167. 

  167.         $retval = false;
    168. 

  168.  
    169. 

  169.         // Get new locale ID, assume default if null
    170. 

  170.         if ($locale != null) {
    171. 

  171.             $locale_id = $locale->getId();
    172. 

  172.             $session->set(SESS_LANGUAGE_VAR, $locale_id);
    173. 

  173.             $retval = true;
    174. 

  174.         } else {
    175. 

  175.             $locale_id = DEFAULT_LANG;
    176. 

  176.             $session->set(SESS_LANGUAGE_VAR, $locale_id);
    177. 

  177.             $retval = false;
    178. 

  178.         }
    179. 

  179.  
    180. 

  180.         if (GETTEXT_AVAILABLE) {
    181. 

  181.             // Try to set locale
    182. 

  182.             $current_locale = setlocale(LC_ALL, $locale_id);
    183. 

  183.  
    184. 

  184.             // Test it against current PHP locale
    185. 

  185.             if ($current_locale != $locale_id) {
    186. 

  186.                 echo "Warning in /classes/Locale.php : Unable to setlocale() to ".$locale_id.", return value: $current_locale, current locale: ".setlocale(LC_ALL, 0);
    187. 

  187.                 $retval = false;
    188. 

  188.             } else {
    189. 

  189.                 bindtextdomain('messages', BASEPATH.'/locale');
    190. 

  190.                 bind_textdomain_codeset('messages', 'UTF-8');
    191. 

  191.                 textDomain('messages');
    192. 

  192.  
    193. 

  193.                 putenv("LC_ALL=".$locale_id);
    194. 

  194.                 putenv("LANGUAGE=".$locale_id);
    195. 

  195.                 $session->set(SESS_LANGUAGE_VAR, $locale_id);
    196. 

  196.                 $retval = true;
    197. 

  197.             }
    198. 

  198.         }
    199. 

  199.  
    200. 

  200.         return $retval;
    201. 

  201.     }

  203.     /**
    204. 

  204.      * Example: 'fr_CA_montreal' will give
    205. 

  205.      * $matches[0]=fr_CA_montreal

  206.      * $matches[1]=fr
    207.

    208.

    210. 

  210.      * wasn't present.

  212.     public static function decomposeLocaleId($locale_id) {
    213. 

  213.         // Init values
    214. 

  214.         $_matches = "";
    215. 

  215.  
    216. 

  216.         $_regex = '/^([^-_]*)(?:[-_]([^-_]*))?(?:[-_]([^-_]*))?$/';
    217. 

  217.         $_match_retval = preg_match($_regex, $locale_id, $_matches);
    218. 

  218.         return $_matches;
    219. 

  219.     }

  221.     /**
    222. 

  222.      * Used by Langstring::GetString() (and other functions) to help select the

  226.     public static function getSqlCaseStringSelect($locale_id) {
    227. 

  227.         $decomposed_locale = Locale :: decomposeLocaleId($locale_id);
    228. 

  228.  
    229. 

  229.         $sql = " (CASE\n";
    230. 

  230.         //On cherche une chaine ou le locale complet correspond au locale par défaut de l'usager
    231. 

  231.  
    232. 

  232.         $sql .= " WHEN locales_id='$decomposed_locale[0]' THEN 1\n";
    233. 

  233.         //On cherche une chaine ou la >langue< du locale correspond a la langue du locale de l'usager (langue générique en premier)
    234. 

  234.         $sql .= " WHEN locales_id='{$decomposed_locale[1]}' THEN 2\n";
    235. 

  235.         //On cherche une chaine ou la >langue< du locale correspond a la langue du locale de l'uager (autres locales de même langue)
    236. 

  236.         $sql .= " WHEN locales_id LIKE '{$decomposed_locale[1]}%' THEN 3\n";
    237. 

  237.         //On cherche une chaine ou le pays du locale correspond au pays du locale de l'uager
    238. 

  238.  
    239. 

  239.         if (!empty ($decomposed_locale[2])) {
    240. 

  240.             $sql .= " WHEN locales_id LIKE '%{$decomposed_locale[2]}' THEN 4\n";
    241. 

  241.         }
    242. 

  242.  
    243. 

  243.         //On cherche une chaine n'ayant pas de locale associée, elle a plue de chance d'être lisible qu'une chaîne prise au hasard
    244. 

  244.         $sql .= " WHEN locales_id IS NULL THEN 5\n";
    245. 

  245.  
    246. 

  246.         $sql .= "      ELSE 20 ";
    247. 

  247.         $sql .= "  END)\n";
    248. 

  248.  
    249. 

  249.         return $sql;
    250. 

  250.     }

    295. 

  295.      * Returns a HTML formatted string for output to string (with an image)

  297.     function GetString() {
    298. 

  298.         // Init values.
    299. 

  299.         $str = "";
    300. 

  300.         $resultats = "";
    301. 

  301.         $preflang_result = "";
    302. 

  302.  
    303. 

  303.         $tmp_loc = $this->GetLocale();
    304. 

  304.  
    305. 

  305.         //Recherche dans la BD du pays et de la langue choisie et affichée selon la langue de préférence de la session
    306. 

  306.         $sql = "SELECT * FROM languages_iso_639_1, locales LEFT JOIN countries ON (locales.countries_id = countries.countries_id) ";
    307. 

  307.         $sql .= "WHERE locales.languages_iso_639_1_id = languages_iso_639_1.iso639_1_id ";
    308. 

  308.         $sql .= "AND locales.locales_id = '$tmp_loc' ";
    309. 

  309.         $this->mBd->ExecuterSqlResUnique($sql, $resultats, FALSE);
    310. 

  310.  
    311. 

  311.         $tmp_pref_lang = $this->mSession->GetPrefLocale();
    312. 

  312.         $sql = "SELECT * FROM locales WHERE locales.locales_id = '$tmp_pref_lang'";
    313. 

  313.         $this->mBd->ExecuterSqlResUnique($sql, $preflang_result, FALSE);
    314. 

  314.  
    315. 

  315.         switch ($preflang_result['languages_iso_639_1_id']) {
    316. 

  316.             case ('fr') :
    317. 

  317.                 {
    318. 

  318.                     $str .= "$resultats[french_name], $resultats[country_french_name]";
    319. 

  319.                     break;
    320. 

  320.                 }
    321. 

  321.             case ('en') :
    322. 

  322.                 {
    323. 

  323.                     $str .= "$resultats[english_name], $resultats[country_english_name]";
    324. 

  324.                     break;
    325. 

  325.                 }

  326.             case ('de') :

  327.                 {
    328. 

  328.                     $str .= "$resultats[german_name], $resultats[country_german_name]";
    329. 

  329.                     break;
    330. 

  330.                 }

  331.             default :
    332. 

  332.                 {
    333. 

  333.                     $str .= "$resultats[french_name], $resultats[country_french_name]";
    334. 

  334.                     break;
    335. 

  335.                 }
    336. 

  336.         }
